        size={18}
        color={currentTheme.secondaryText}
      />
    </TouchableOpacity>

    {/* Expandable vault list */}
    {isVaultSwitcherVisible && (
      <View style={{ marginLeft: 8 }}>
        {vaults.map((vault) => (
          <TouchableOpacity
            key={vault.id}
            style={[
              styles.sidebarItem,
              activeVaultId === vault.id && styles.activeItem,
            ]}
            onPress={async () => {
              // Auto-save current note before switching
              if (isEditing) {
                await handleSaveNote();
              }
              setIsVaultSwitcherVisible(false);
              await handleSetActiveVault(vault.id);
              // Reload notes from new vault
              const vaultNotes = await getAllNotesFromFilesystem(vault.id);
              setNotes(vaultNotes);
            }}
          >
            <MaterialCommunityIcons
              name={activeVaultId === vault.id ? 'safe-square' : 'safe-outline'}
              size={18}
              color={activeVaultId === vault.id ? currentTheme.primary : currentTheme.secondaryText}
            />
            <Text
              style={[
                styles.sidebarText,
                { flex: 1 },
                activeVaultId === vault.id && { color: currentTheme.primary },
              ]}
              numberOfLines={1}
            >
              {vault.name}
            </Text>
            {/* Delete vault (only if >1 vault exists and it's not the currently active) */}
            {vaults.length > 1 && activeVaultId !== vault.id && (
              <TouchableOpacity
                onPress={() => {
                  setRenameVaultId(vault.id);
                  setRenameVaultName(vault.name);
                  setIsRenameVaultModalVisible(true);
                }}
                style={{ padding: 4, marginRight: 4 }}
              >
                <MaterialCommunityIcons name="pencil" size={16} color={currentTheme.secondaryText} />
              </TouchableOpacity>
            )}
            {vaults.length > 1 && (
              <TouchableOpacity
                onPress={() => {
                  showConfirmation(
                    `Delete vault "${vault.name}"? All notes inside will be permanently deleted.`,
                    async () => {
                      await deleteVault(vault.id);
                      const remaining = await getVaults();
                      setVaults(remaining);
                      // If deleted vault was active, handled by deleteVault
                      const newActive = await getActiveVaultId();
                      if (newActive && newActive !== activeVaultId) {
                        await handleSetActiveVault(newActive);
                        const vaultNotes = await getAllNotesFromFilesystem(newActive);
                        setNotes(vaultNotes);
                      }
                    }
                  );
                }}
                style={{ padding: 4 }}
              >
                <MaterialCommunityIcons name="delete-outline" size={16} color="#FF5252" />
              </TouchableOpacity>
            )}
          </TouchableOpacity>
        ))}

        {/* Create Vault button */}
        <TouchableOpacity
          style={[styles.sidebarItem, { opacity: 0.7 }]}
          onPress={() => setIsCreateVaultModalVisible(true)}
        >
          <MaterialCommunityIcons name="plus-circle-outline" size={18} color={currentTheme.primary} />
          <Text style={[styles.sidebarText, { color: currentTheme.primary }]}>New Vault</Text>
        </TouchableOpacity>

        {/* Import .md files button */}
        <TouchableOpacity
          style={[styles.sidebarItem, { opacity: 0.7 }]}
          onPress={async () => {
            setIsImporting(true);
            try {
              const vault = await importMDFilesAsVault();
              if (vault) {
                const updatedVaults = await getVaults();
                setVaults(updatedVaults);
                await handleSetActiveVault(vault.id);
                const vaultNotes = await getAllNotesFromFilesystem(vault.id);
                setNotes(vaultNotes);
                showConfirmation(
                  `Imported vault "${vault.name}" with markdown files.`,
                  () => {}
                );
              }
            } catch (err) {
              console.error('Import failed:', err);
              showConfirmation('Import failed. Please try again.', () => {});
            } finally {
              setIsImporting(false);
            }
          }}
        >
          <MaterialCommunityIcons name="file-import-outline" size={18} color={currentTheme.primary} />
          <Text style={[styles.sidebarText, { color: currentTheme.primary }]}>
            {isImporting ? 'Importing...' : 'Import .md Files'}
          </Text>
        </TouchableOpacity>
      </View>
    )}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User Input ↔ Vault Modals | Vault names entered in create/rename modals could be empty or contain special characters |
| UI State ↔ AsyncStorage | Optimistic UI updates could get out of sync if AsyncStorage writes fail silently |
| vaultImport ↔ Filesystem | Imported files could have names causing filesystem conflicts |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-06 | Tampering | Vault name input | mitigate | Vault names are stored as labels in AsyncStorage, never used in filesystem paths (vault IDs are). Empty names are rejected at input. |
| T-08-07 | Denial of Service | Vault list UI with 100+ vaults | accept | Practical limit for local vaults is ~10-20. FlatList would handle 100+. Not a realistic DoS vector for local app. |
| T-08-08 | Repudiation | Vault deletion | accept | Deletion requires explicit user confirmation via the confirmation dialog. Undo not implemented (Obsidian doesn't have vault undo either). |
| T-08-09 | Information Disclosure | Vault switcher shows all vault names | accept | Vault names are local-only. No sensitive data is leaked. User created them. |
</threat_model>
