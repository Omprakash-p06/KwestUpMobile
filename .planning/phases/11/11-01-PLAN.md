  <action>
    Create `src/utils/exportService.js` with the following:

    1. **Imports:**
       ```javascript
       import CryptoJS from "crypto-js";
       import * as FileSystem from "expo-file-system";
       import * as Sharing from "expo-sharing";
       import AsyncStorage from "@react-native-async-storage/async-storage";
       import { isUserDataKey, APP_VERSION, STORAGE_VERSION } from "./storage";
       import { getVaults, getVaultPath } from "./vaultService";
       ```

    2. **encryptBackup(payloadObj, passphrase):**
       - JSON.stringify the payload
       - CryptoJS.AES.encrypt(rawText, passphrase).toString()
       - Wrap in try/catch, throw on failure

    3. **decryptBackup(encryptedText, passphrase):**
       - CryptoJS.AES.decrypt(encryptedText, passphrase)
       - bytes.toString(CryptoJS.enc.Utf8)
       - If empty string → throw Error("Invalid password or corrupted file.")
       - JSON.parse the decrypted text
       - Wrap in try/catch, throw on failure with message "Unable to decrypt archive. Please verify the passphrase."

    4. **packVaults():**
       - Call getVaults() to get vault list
       - For each vault, getVaultPath(vault.id) then FileSystem.getInfoAsync
       - If exists, readDirectoryAsync, recurse one level deep for folders
       - Read .md files via FileSystem.readAsStringAsync
       - Push { folder, title (without .md), content }
       - Return array of { id, name, createdAt, updatedAt, notes }
       - Use Promise.all for files within each folder (per pitfall: large vault perf)

    5. **collectAsyncStorageData():**
       - AsyncStorage.getAllKeys()
       - Filter with isUserDataKey(key) — only user data keys
       - AsyncStorage.multiGet(userKeys) 
       - Return object: { [key]: value } for all user data keys

    6. **exportArchive(passphrase, onProgress):**
       Async function:
       - Call onProgress(0.1) — starting
       - Run collectAsyncStorageData() → storageData
       - Call onProgress(0.3) — storage collected
       - Run packVaults() → vaultData
       - Call onProgress(0.6) — vaults packed
       - Build payload:
         ```javascript
         const payload = {
           metadata: {
             version: "1.0",
             timestamp: new Date().toISOString(),
             appVersion: APP_VERSION,
             storageVersion: STORAGE_VERSION,
           },
           storage: storageData,
           vaults: vaultData,
         };
         ```
       - encryptBackup(payload, passphrase)
       - Call onProgress(0.8) — encrypted
       - Write to FileSystem.cacheDirectory + "kwestup-backup.kwestup"
       - Call onProgress(0.9) — written
       - Sharing.shareAsync(filePath, { mimeType: "application/octet-stream", dialogTitle: "Export KwestUp Backup" })
       - Delete the cache file after sharing: FileSystem.deleteAsync(filePath, { idempotent: true })
       - Call onProgress(1.0) — done
       - Wrap entire function in try/catch, always clean up cache file in finally block
  </action>

    **Step 3: Add handler functions (before the return statement)**

    ```javascript
    const handleExportPress = () => {
      setExportPassphrase("");
      setConfirmPassphrase("");
      setPassphraseError(null);
      setArchiveModalMode("export");
      setArchiveModalVisible(true);
    };

    const handleImportPress = async () => {
      try {
        const result = await DocumentPicker.getDocumentAsync({
          type: "*/*",
          copyToCacheDirectory: true,
        });
        if (!result.canceled && result.assets?.length > 0) {
          setSelectedArchiveFile(result.assets[0]);
          setImportPassphrase("");
          setPassphraseError(null);
          setArchiveModalMode("import");
          setArchiveModalVisible(true);
        }
      } catch (err) {
        setArchiveError("Failed to open file picker. Please try again.");
      }
    };

    const handleExportConfirm = async () => {
      if (exportPassphrase !== confirmPassphrase) {
        setPassphraseError("PASSPHRASES DO NOT MATCH");
        return;
      }
      if (!exportPassphrase) {
        setPassphraseError("PASSPHRASE CANNOT BE EMPTY");
        return;
      }
      setArchiveModalVisible(false);
      setArchiveProcessing(true);
      setArchiveProgress(0);
      setArchiveError(null);
      try {
        await exportArchive(exportPassphrase, (progress) => {
          setArchiveProgress(progress);
        });
      } catch (err) {
        setArchiveError(err.message || "ARCHIVE ERROR: Failed to export.");
      } finally {
        setArchiveProcessing(false);
        setArchiveProgress(0);
      }
    };

    const handleModalCancel = () => {
      setArchiveModalVisible(false);
      setPassphraseError(null);
      setExportPassphrase("");
      setConfirmPassphrase("");
      setImportPassphrase("");
    };
    ```
